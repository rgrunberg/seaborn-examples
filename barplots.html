Messing with the axes 

plot = sns.barplot(data=tips, x='day', y='total_bill', hue='time')
plot.set_title('A barplot with some data')
plot.set_xlabel('The X axis label')
plot.set_ylabel('The Y axis has 16 ticks')

# change the number of axis ticks
plot.yaxis.set_major_locator(plt.MaxNLocator(16))

# format the axis labels
plot.yaxis.set_major_formatter(ticker.FuncFormatter(lambda x, _: '${:.2f}'.format(x)))

# rotate the axis labels
plot.set_xticklabels(plot.get_xticklabels(), rotation=45);

plot = sns.barplot(data=tips, x='day', y='total_bill', hue='time',

plot.axhline(10, lw=1.75, color='red')

# draw a horizontal line behind the bars
plot.axhline(5, lw=1.75, color='red').set_zorder(0);
